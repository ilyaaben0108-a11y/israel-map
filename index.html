// פונקציה שבודקת אם יש התרעות חדשות async function checkAlerts() { try { // זהו API לדוגמה - תצטרך להחליף אותו במקור הנתונים שלך const response = await fetch('https://api.redalert.me/alerts'); const alerts = await response.json(); if (alerts && alerts.length > 0) { alerts.forEach(alert => { // נניח וה-API מחזיר קואורדינטות [קו רוחב, קו אורך] displayAlertOnMap(alert.lat, alert.lng, alert.city); }); } } catch (error) { console.error("שגיאה במשיכת התרעות:", error); } } // פונקציה שמציירת את ההתרעה על המפה function displayAlertOnMap(lat, lng, cityName) { const alertCircle = L.circle([lat, lng], { color: 'red', fillColor: '#f03', fillOpacity: 0.8, radius: 5000 // רדיוס של 5 קילומטר }).addTo(map); alertCircle.bindPopup("צבע אדום: " + cityName).openPopup(); // הסרת ההתרעה אחרי 30 שניות כדי שהמפה לא תתמלא setTimeout(() => { map.removeLayer(alertCircle); }, 30000); } // הפעלת הבדיקה בכל 5 שניות setInterval(checkAlerts, 5000);
